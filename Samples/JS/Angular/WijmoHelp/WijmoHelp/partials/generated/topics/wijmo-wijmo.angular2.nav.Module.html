wijmo/wijmo.angular2.nav

Contains Angular 2 components for the wijmo.nav module.

wijmo.angular2.nav is an external TypeScript module that can be imported to your code using its ambient module name. For example:

import * as wjNav from 'wijmo/wijmo.angular2.nav';
 
@Component({
    directives: [wjNav.WjTreeView],
    template: `
      <wj-tree-view [itemsSource]="items" [displayMemberPath]="'header'" [childItemsPath]="'items'">
      </wj-tree-view;`,
    selector: 'my-cmp',
})
export class MyCmp {
    data: any[];
}
